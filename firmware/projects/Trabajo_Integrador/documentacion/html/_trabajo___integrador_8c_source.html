79float calcular_desvio_estandar(float *vector_datos, int longitud)
80{
81 float media = 0.0;
82 float sumatoria_cuadrados = 0.0;
83
84 // Calcular la media de los datos
85 for (int i = 0; i < longitud; i++)
86 {
87 media += vector_datos[i];
88 }
89 media /= longitud;
90 // Calcular la sumatoria de los cuadrados de las diferencias
91 for (int i = 0; i < longitud; i++)
92 {
93 sumatoria_cuadrados += powf(vector_datos[i] - media, 2);
94 }
95 // Calcular el desvío estándar
96 float desvio_estandar = sqrtf(sumatoria_cuadrados / longitud);
97 return desvio_estandar;
98}
