57void distanciaTask(void *pvParameter){
58 uint16_t distancia;
59 while(1)
60 {
61 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */
62 if(start)
63 {
64 distancia=HcSr04ReadDistanceInCentimeters();
65 UartSendString(UART_PC,(char*)UartItoa(distancia,10));
66 UartSendString(UART_PC,"cm \r\n");
67 if(!hold)
68 {
69 LcdItsE0803Write(distancia);
70 }
71 if(distancia<10){
72 LedOff(LED_1);
73 LedOff(LED_2);
74 LedOff(LED_3);
75 }
76 else if(distancia<20){
77 LedOn(LED_1);
78 LedOff(LED_2);
79 LedOff(LED_3);
80 }
81 else if(distancia<30){
82 LedOn(LED_1);
83 LedOn(LED_2);
84 LedOff(LED_3);

88 LedOn(LED_2);
