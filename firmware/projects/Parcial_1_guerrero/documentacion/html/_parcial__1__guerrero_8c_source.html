100void senial_medir_task(void *pvParameter)
101{
102 while (1)
103 {
104 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
105 if (start)
106 {
107 // obtengo el valor del ph -> con esto veo si es menor o mayor y activo su respectiva bomba
108 AnalogInputReadSingle(CH1, &valor_ph);
109 float voltage = (valor_ph / TOTAL_BITS) * 3.0; // Convertir a voltaje (0-3V)
110 ph = (voltage / V_REF) * 14.0; // Convertir a pH (0-14)
111 // obtengo el valor de la humedad
112 AnalogInputReadSingle(CH2, &valor_humedad);
113 conversion_humedad = valor_humedad * (V_REF / TOTAL_BITS); // valor medido a voltios
114 }
115 }
116}
120void Informar_estado_task(void *pvParameter)
121{
122 while (1)
123 {
124 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
125 if (hold)
126 {
127 if (ph < 6.0)
128 {
129 encender_SolBasica = true;

134 UartSendString(UART_PC, "Enciendo bomba de solucion basica \r\n");
135 }
136 else if (ph > 6.7)
137 {
138 encender_SolAcida = true;
139 GPIOOn(GPIO_2);
140 UartSendString(UART_PC, (char *)UartItoa(valor_ph, 10));
141 UartSendString(UART_PC, "El ph esta basico \r\n");
142 UartSendString(UART_PC, (char *)UartItoa(valor_ph, 10));
143 UartSendString(UART_PC, "Enciendo bomba de solucion acida \r\n");
144 }
145 else
146 {
147 encender_SolAcida = false;
148 encender_SolBasica = false;
149 GPIOOff(GPIO_2);
150 GPIOOff(GPIO_3);
151 UartSendString(UART_PC, (char *)UartItoa(ph, 10));
152 UartSendString(UART_PC, "El ph esta optimo \r\n");
153 }
154
155 if (conversion_humedad < valor_humedad_umbral)
156 {
157 encender_bomba_agua = true;
158 GPIOOn(GPIO_1);
159 UartSendString(UART_PC, (char *)UartItoa(ph, 10));
160 UartSendString(UART_PC, "Bomba de agua encendida \r\n");

162 else
